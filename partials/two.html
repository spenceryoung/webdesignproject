AngularJS

AngularJS is a great solution for data-binding and routing

AngularJS is an MVC (Model View Controller) JavaScript framework. It holds the concept of single page web applications by updating views instead of preforming a full browser refresh. As one might guess, this concept significantly increases performance and decreases repetition of HTML elements.

            app.config([
                '$routeProvider',
                function($routeProvider) {
                    $routeProvider
                    .when('/home', {
                        templateUrl: 'partials/one.html',
                        controller: 'OneCtrl'
                    })
                    .when('/angularjs', {
                        templateUrl: 'partials/two.html',
                        controller: 'TwoCtrl'
                    })
                    .when('/jquery', {
                        templateUrl: 'partials/three.html',
                        controller: 'ThreeCtrl'
                    })
                    .when('/css3', {
                        templateUrl: 'partials/four.html',
                        controller: 'FourCtrl'
                    })
                    .when('/mithril', {
                        templateUrl: 'partials/five.html',
                        controller: 'FiveCtrl'
                    })
                    .otherwise({
                        redirectTo: '/home'
                    });
                }
            ]);
